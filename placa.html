import cv2
import pytesseract
pytesseract.pytesseract.tesseract_cmd=r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe"

#listas de placa aceptadas en el estacionamiento del CBTis 144
placas_autorizadas=["XYZ789","RSS1536","7ADB538","AA-000-AA"]
placas_no_autorizadas=["AS-852-XZ","AS52 RXZ"]

#cargar placa a leer
placa_xy=cv2.imread("placas.jpg")
placa_a1=cv2.imread("PLACA.png")
placa_real_1=cv2.imread("Placas_3.jpg")
placa_real_2=cv2.imread("placa_4.jpg")
placa_no_autorizada=cv2.imread("placa_no_atorizada.jpg")
placa_2_no=cv2.imread("placa_no_2.jpg")

placanueva=cv2.resize(placa_xy,(800,300))
placa_nuevo=cv2.resize(placa_a1,(800,300))
placa_real1=cv2.resize(placa_real_1,(800,300))
placa_real2=cv2.resize(placa_real_2,(800,300))
placa_no=cv2.resize(placa_no_autorizada,(800,300))
placa_no_2=cv2.resize(placa_2_no,(800,300))

placa_gris=cv2.cvtColor(placanueva,cv2.COLOR_BGR2GRAY)
placa_2gris=cv2.cvtColor(placa_nuevo,cv2.COLOR_BGR2GRAY)
placa_real1_gris=cv2.cvtColor(placa_real1,cv2.COLOR_BGR2GRAY)
placa_real2_gris=cv2.cvtColor(placa_real2,cv2.COLOR_BGR2GRAY)
placano=cv2.cvtColor(placa_no_autorizada,cv2.COLOR_BGR2GRAY)
placa_noautorizada_2=cv2.cvtColor(placa_no_2,cv2.COLOR_BGR2GRAY)

plca_res=cv2.threshold(placa_gris,120,255,cv2.THRESH_BINARY)[1]
placa_res=cv2.threshold(placa_2gris,120,255,cv2.THRESH_BINARY)[1]
placa_real1_res=cv2.threshold(placa_real1_gris,120,255,cv2.THRESH_BINARY)[1]
placa_real2_res=cv2.threshold(placa_real2_gris,120,255,cv2.THRESH_BINARY)[1]
placa_no_res=cv2.threshold(placano,120,255,cv2.THRESH_BINARY)[1]
placa_2_res=cv2.threshold(placa_noautorizada_2,120,255,cv2.THRESH_BINARY)[1]

texto_placa=pytesseract.image_to_string(plca_res,lang="spa")
texto_placa2=pytesseract.image_to_string(placa_res,lang="spa")
texto_placa_real1=pytesseract.image_to_string(placa_real1_res,lang="spa")
texto_placa_real2=pytesseract.image_to_string(placa_real2_res,lang="spa")
texto_placano=pytesseract.image_to_string(placa_no_res,lang="spa")
texto_placano2=pytesseract.image_to_string(placa_2_res,lang="spa")

texto_limpio=texto_placa.strip().replace(" ","")
texto_limpio2=texto_placa2.strip().replace(" ","")
texto_limpio_real1=texto_placa_real1.strip().replace(" ","")
texto_limpio_real2=texto_placa_real2.strip().replace(" ","")
texto_no_limpio=texto_placano.strip().replace(" ","")
texto_no_limpio2=texto_placano2.strip().replace(" ","")

#buscar la placa en la lista autorizada
if texto_limpio in placas_autorizadas:
    print("OKEY!!,\nBienvenido al estacionamiento del CBTis 144")
elif texto_limpio2 in placas_autorizadas:
    print("OKEY!!,\nBienvenido al estacionamiento del CBTis 144")
elif texto_limpio_real1 in placas_autorizadas:
    print("OKEY!! \nBienvenido al estacionamiento del CBTis 144")
elif texto_limpio_real2 in placas_autorizadas:
    print("OKEY!! \nBienvenido al estacionamiento del CBTis 144")
elif texto_no_limpio in placas_no_autorizadas:
    print("OKEY!! \nNo tienes permitido pasar al estacionamiento del CBTis 144")
elif texto_no_limpio2 in placas_no_autorizadas:
    print("OKEY!!! \nNo tienes permitido pasar al estacionamiento del CBTis 144")


print(texto_limpio)
print(texto_limpio2)
print(texto_limpio_real1)
print(texto_limpio_real2)
print(texto_no_limpio)
print(texto_no_limpio2)

print("Autorizadas")

print(texto_placa)
print(texto_placa2)

cv2.imshow("imagen 2 analizada",placa_res)
cv2.imshow("imagen real analizada",placa_real1_res)
cv2.imshow("imagen real 2 analizada",placa_real2_res)
cv2.imshow("imagen de placa no permitida",placa_no_res)

imsho

waitKey

sirve para esperar una tecla del teclado mientras espera una imagen

destroyAllWindows

cierra la ventana cuyo nombre pongas entre comillas

COLOR_BGR2GRAY

sirve para convertir una imagen a blanco y negro (escala de grises) en OpenCV.
